Grammar:

UpperCase 'V', Vertical Bar '|', and Comma ',' == OR
Ampersand '&', Asterisk '*', and Colon ':' == AND
Question Mark '?', Exlamation Point '!', and Pound '#' == NOT
Open Parenthesis '(', Open Bracket '[', and Less Than '<' == CLAUSE START/OPEN CLAUSE
Close Parenthesis ')', Close Bracket ']', and Greater Than '>' == CLAUSE END/CLOSE CLAUSE
